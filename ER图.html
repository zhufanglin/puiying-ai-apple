    roles {
        int       id            PK  "自增主键"
        string    name          UK  "角色名 super_admin / principal / ..."

        datetime  created_at         "创建时间"
        datetime  updated_at         "更新时间"
    }

    users {
        int       id            PK  "自增主键"
        string    username      UK  "登录名"
        string    password_hash      "bcrypt 哈希"
        string    display_name       "显示姓名"
        int       role_id       FK  "关联 roles.id"
        string    phone              "手机号"
        string    email              "邮箱"
        bool      is_active          "是否启用"
        datetime  created_at         "创建时间"
        datetime  updated_at         "更新时间"
    }

    permissions {
        int       id            PK  "自增主键"
        string    code          UK  "权限码 apple:awards:read"
        string    name               "权限名称"
        string    module             "归属模块 apple / accounts / admin"
        datetime  created_at         "创建时间"
        datetime  updated_at         "更新时间"
    }

    role_permissions {
        int       role_id       PK_FK "关联 roles.id"
        int       permission_id PK_FK "关联 permissions.id"
    }

    files {

        bigint    size_bytes         "文件大小(字节)"

        int       entity_id          "关联实体ID"

        datetime  updated_at         "更新时间"
    }

    ocr_jobs {
        int       id            PK  "自增主键"

        jsonb     result_json        "OCR 结构化结果"
        string    error_message      "失败原因"
        int       created_by    FK  "提交者 users.id"
        datetime  created_at         "创建时间"
        datetime  updated_at         "更新时间"
    }

    ai_jobs {
        int       id            PK  "自增主键"

        jsonb     output_json        "AI 结构化结果"
        string    error_message      "失败原因"
        int       created_by    FK  "提交者 users.id"
        datetime  created_at         "创建时间"
        datetime  updated_at         "更新时间"
    }

    audit_logs {

        string    ip_address         "操作IP"
