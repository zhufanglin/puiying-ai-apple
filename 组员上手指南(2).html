1. 项目是什么

这是培英中学的 AI 数智化管理平台，第一个子系统叫 Apple，包含 4 个模块：

🏆 A1 奖状奖学金

模板管理 → AI 生成评语 → 审批 → 打印

💰 A2 财务收支

收支登记 → 拍照 OCR 识别票据 → 统计

📦 A3 资产盘点

资产入库 → 盘点 → 折旧 → 报废审批

🎓 A4 学生事务

学生档案 → AI 画像 → 在校证明 → 请假

技术栈：后端 Python + FastAPI，前端 Next.js + React + Tailwind，数据库 PostgreSQL，队列 Redis + Celery，OCR 用 PaddleOCR。

5. 框架怎么用（核心规则）

5.1 API 返回格式——全部统一

所有接口返回的 JSON 格式必须是这样：

// 成功
{"code": 0, "message": "ok", "data": { ... }}

// 失败
{"code": 20001, "message": "资源不存在", "data": null}

回到你的路由函数里写：

from app.common.schemas import APIResponse

return APIResponse(data={...})          # 成功
return APIResponse(code=20001, message="没找到")  # 失败

5.2 数据库——用 Depends 注入

在路由参数里加一行，框架自动给连接、自动 commit、自动 close：

from fastapi import Depends
from sqlalchemy.ext.asyncio import AsyncSession
from app.db.session import get_db

@router.get("/items")
async def list_items(db: AsyncSession = Depends(get_db)):
    result = await db.execute(select(MyModel))
    items = result.scalars().all()
    return APIResponse(data=items)

5.3 权限——加一行装饰器

from app.core.security import get_current_user
from app.core.permissions import require_permission, Permissions

@router.get("/awards")
async def list_awards(
    db: AsyncSession = Depends(get_db),
    user = Depends(get_current_user),                          # 拿当前用户
    _ = Depends(require_permission(Permissions.AWARDS_READ)),  # 检查权限
):
    ...

from app.common.errors import NOT_FOUND, CONFLICT, raise_error

if not item:

6. 后端开发模板（照着写）

以 A1 奖状模块为例，你需要创建 3 个文件：

① models.py —— 数据库表

# apps/api/app/modules/apple/awards/models.py
from typing import Optional
from sqlalchemy import String, Text, ForeignKey
from sqlalchemy.orm import Mapped, mapped_column
from app.db.base import Base, TimestampMixin

class Award(Base, TimestampMixin):
    __tablename__ = "awards"

    name: Mapped[str] = mapped_column(String(200), comment="奖状名称")
    template_path: Mapped[str] = mapped_column(String(500), comment="模板路径")
    created_by: Mapped[int] = mapped_column(comment="创建人")
    # id / created_at / updated_at 自动继承自 TimestampMixin

② router.py —— API 路由

# apps/api/app/modules/apple/awards/router.py
from fastapi import APIRouter, Depends
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from app.common.schemas import APIResponse
from app.common.errors import NOT_FOUND, raise_error
from app.core.security import get_current_user
from app.core.permissions import require_permission, Permissions
from app.db.session import get_db
from app.modules.accounts.models import User
from .models import Award

router = APIRouter()

@router.get("")
async def list_awards(
    db: AsyncSession = Depends(get_db),
    user = Depends(get_current_user),
    _ = Depends(require_permission(Permissions.AWARDS_READ)),
    page: int = 1, page_size: int = 20,
):
    result = await db.execute(
        select(Award).offset((page-1)*page_size).limit(page_size)
    )
    items = result.scalars().all()
    return APIResponse(data={ "items": items, "page": page })

@router.get("/{award_id}")
async def get_award(award_id: int, db: AsyncSession = Depends(get_db)):
    result = await db.execute(select(Award).where(Award.id == award_id))
    item = result.scalar_one_or_none()
    if not item:
        raise_error(*NOT_FOUND)
    return APIResponse(data=item)

③ 注册路由到 main.py

写完 router.py 后，去 apps/api/app/main.py 找到对应注释，取消注释：

# 原来是被注释掉的：
# from app.modules.apple.awards.router import router as awards_router
# app.include_router(awards_router, prefix=..., tags=["Apple-奖状奖学金"])

# 取消注释（去掉开头的 #）变成：
from app.modules.apple.awards.router import router as awards_router
app.include_router(awards_router, prefix="/api/v1/apple/awards", tags=["Apple-奖状奖学金"])

7. 前端开发模板（照着写）

列表页示例（替换占位页面）

// apps/web/app/(dashboard)/dashboard/apple/awards/page.tsx
"use client";

import { useState, useEffect } from "react";
import { Plus } from "lucide-react";
import PageHeader from "@/components/ui/PageHeader";
import DataTable, { Column } from "@/components/ui/DataTable";
import FilterBar from "@/components/ui/FilterBar";
import StatsCard from "@/components/ui/StatsCard";
import EmptyState from "@/components/ui/EmptyState";
import { api } from "@/lib/api";
import { Award } from "lucide-react";

interface AwardItem {
  id: number; name: string; template_path: string; created_at: string;
}

const columns: Column<AwardItem>[] = [
  { key: "id", header: "ID", width: "80px" },
  { key: "name", header: "奖状名称" },
  { key: "created_at", header: "创建时间" },
];

export default function AwardsPage() {
  const [data, setData] = useState<AwardItem[]>([]);
  const [total, setTotal] = useState(0);
  const [loading, setLoading] = useState(true);

  const fetchData = async (page = 1) => {
    setLoading(true);
    const res = await api.get<{ items: AwardItem[]; total: number }>(
      `/apple/awards?page=${page}&page_size=20`
    );
    setData(res.data?.items || []);
    setTotal(res.data?.total || 0);
    setLoading(false);
  };

  useEffect(() => { fetchData(); }, []);

  return (
    <div>
      <PageHeader
        title="奖状奖学金"
        subtitle="管理奖状模板、生成证书、审批流程"
        actions={<button className="px-4 py-2 bg-blue-500 text-white rounded-lg">
          <Plus size={16} /> 新建奖状
        </button>}
      />

      <div className="grid grid-cols-3 gap-4 mb-6">
        <StatsCard label="奖状模板" value={total} icon={Award} color="text-blue-600" />
      </div>

      {data.length > 0 ? (
        <DataTable columns={columns} data={data} total={total} loading={loading} />
      ) : (
        <EmptyState icon={Award} title="暂无奖状" description="点击右上角新建" />
      )}
    </div>
  );
}

弹窗组件示例（放在 components/modules/apple/）

// apps/web/components/modules/apple/AwardsScriptDialog.tsx
"use client";

import { useState } from "react";
import { X } from "lucide-react";
import AiReviewPanel from "@/components/ui/AiReviewPanel";
import { api } from "@/lib/api";

interface Props { awardId: number; onClose: () => void; }

export default function AwardsScriptDialog({ awardId, onClose }: Props) {
  const [aiContent, setAiContent] = useState("");
  const [loading, setLoading] = useState(false);

  const handleGenerate = async () => {
    setLoading(true);
    const res = await api.post<{ text: string }>("/apple/awards/generate-script", { awardId });
    setAiContent(res.data?.text || "");
    setLoading(false);
  };

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/30">
      <div className="bg-white rounded-xl p-6 w-full max-w-lg">
        <div className="flex items-center justify-between mb-4">
          <h3 className="text-lg font-semibold">AI 奖状评语</h3>
          <button onClick={onClose}><X size={20} /></button>
        </div>
        <button onClick={handleGenerate} className="w-full py-2 bg-purple-500 text-white rounded-lg">
          生成 AI 评语
        </button>
        {aiContent && <AiReviewPanel content={aiContent} loading={loading} />}
      </div>
    </div>
  );
}

9. 常见问题

Q：Docker 启动后数据库连不上？
A：等 10 秒，PostgreSQL 需要初始化。用 docker-compose ps 确认状态是 "Up (healthy)" 而不是 "starting"。
Q：pip install 报错？
A：确保已经激活虚拟环境（终端前面有 (.venv)）。如果是 PaddleOCR 装不上，先跳过 OCR 功能，用 mock 数据。
Q：路由注册后 404？
A：检查 main.py 里对应那行注释是否已取消。重启 uvicorn。
Q：前端 API 调用报 CORS 错误？
A：开发期 CORS 已全开，检查 NEXT_PUBLIC_API_URL 是否正确（默认 http://localhost:8000）。
Q：我不确定某个功能怎么写？
A：1) 先看本文档的模板。2) 看 apps/api/app/modules/accounts/models.py 和 router.py 作为参考示例。3) 问同学1。
Q：种子数据里管理员账号是什么？
A：admin / admin123。可以用这个账号登录测试。
Q：我改了 core/ 或 db/ 里的代码，全坏了怎么办？
A：git checkout -- apps/api/app/core/ 恢复。千万不要改底座代码！
